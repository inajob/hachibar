## これはなに? HACHIBARは手のひらサイズのモバイルゲーム機です。 Arduinoを基にしているので、Arduinoを使ってオリジナルのゲームを作ることができます。 現在絶賛開発中です。 ## スペック - ディスプレイ: 16×8ドットマトリクスLED - CPU: ATmega328 (ROM: 32K, SRAM 2K) - 小型スピーカー - 十字キー + A,Bボタン - 電源: CR2032 ## サンプルプログラム キーの入力を画面に表示する例です。 HACHIBAR専用のSDKが提供するHachibarオブジェクトを通じてディスプレイへの出力や、キー入力の読み込みができます。 
#include "hachibar.h"

Hachibar h;

void setup(){
  h.init();
}
void loop(){
  h.rect(false, 0, 0, 8, 16);
  if(h.trigger[h.BTN_A] > 3){
    h.drawChar('A', 2, 2);
  }
  if(h.trigger[h.BTN_B] > 3){
    h.drawChar('B', 2, 2);
  }

  if(h.trigger[h.BTN_UP] > 3){
    h.drawChar('^', 2, 2);
  }
  if(h.trigger[h.BTN_DOWN] > 3){
    h.drawChar('V', 2, 2);
  }
  if(h.trigger[h.BTN_LEFT] > 3){
    h.drawChar('<', 2, 2);
  }
  if(h.trigger[h.BTN_RIGHT] > 3){
    h.drawChar('>', 2, 2);
  }
  h.updateKey();
  h.updateDisplay();
}
